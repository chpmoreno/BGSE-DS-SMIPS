Exercise 2.1¶

If $t = 1$

\begin{align} p(t=1|\textbf{x}) &= \frac{1}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}} \\ & =\frac{1}{1+\exp^{-t(\text{w}^T\Phi(\textbf{x})+b)}} \end{align}

If $t = -1$

\begin{align} p(t=-1|\textbf{x}) &= 1-\frac{1}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}}\\ &=\frac{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}}-\frac{1}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}}\\ &=\frac{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}-1}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}} \\ &=\frac{\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}}{1+\exp^{-(\text{w}^T\Phi(\textbf{x})+b)}} \\ &=\frac{1}{1+\exp^{\text{w}^T\Phi(\textbf{x})+b}} \\ &=\frac{1}{1+\exp^{-t(\text{w}^T\Phi(\textbf{x})+b)}} \end{align}

Hence:

\begin{equation} p(t|\textbf{x})=\frac{1}{1+\exp^{-t(\text{w}^T\Phi(\textbf{x})+b)}} \end{equation}

Exercise 2.2¶

We are going to minimise:

\begin{equation} \frac{1}{N}\sum_n L \left(t_n(\Phi(\textbf{x}_n)^T\textbf{w}+b)\right)+\frac{\lambda}{2}\textbf{w}^T\textbf{w} \end{equation}

Where the loss function $L(u)=\text{log}(1+\exp^{-u})$. Therefore the large margin $w$ is:

\begin{equation} \text{w}_{LM}=\frac{1}{N}\sum_n \log\left(1+\exp^{-t_n(\Phi(\textbf{x}_n)^T\textbf{w}+b)}\right)+\frac{\lambda}{2}\textbf{w}^T\textbf{w} \end{equation}

Then similar as logistic regression likelihood with gaussian prior:

\begin{equation} p(\textbf{w}|\textbf{t})\propto p(\textbf{w})p(\textbf{t}|\textbf{w}) \end{equation}

We have:

\begin{align} \log (p(\textbf{w}))&=-\frac{1}{2}(\textbf{w}-\textbf{0})^T\lambda N(\textbf{w}-\textbf{0}) \\ &=-\frac{1}{2}\lambda N\textbf{w}^T\textbf{w} \end{align}\begin{align} \log (p(\textbf{t}|\textbf{w}))&=\sum_n \log\left(\dfrac{1}{1+\exp^{-\textbf{t}_n(\Phi(x_n)^T\textbf{w}+b)}}\right) \\ & =\sum_n-\log\left(1+\exp^{-\textbf{t}_n(\Phi(x_n)^T\textbf{w}+b)}\right) \end{align}

So:

$$\log (p(\textbf{w}|\textbf{t}))=\sum_n-\log \left(1+\exp^{-\textbf{t}_n(\Phi(x_n)^T\textbf{w}+b)}\right)-\frac{1}{2}\lambda N\textbf{w}^T\textbf{w} + c$$

The stationary point of this (w.r.t w) will be the same as for the large margin classifier.

Exercise 5¶

In robust regression model we consider a Gamma prior $p(\eta_n)=\text{Gamma}\left(\frac{\nu}{2},\frac{\nu}{2}-1\right)$ and a normal likelihood $p(\mathbf{t}|\mathbf{X},\mathbf{w},q,\mathbf{\eta})=\mathcal{N}(\mathbf{\Phi w},(q\text{diag}(\mathbf{\eta}))^{-1})$. To compute the posterior we use Bayes theorem, obtaining

\begin{equation} p(\eta_n|t_n,\nu)=\frac{p(t_n|\eta_n,\nu)p(\eta_n|\nu)}{p(t_n|\nu)}\propto p(t_n|\eta_n,\nu)p(\eta_n|\nu)= \text{Gamma}\left(\frac{\nu}{2},\frac{\nu}{2}-1\right)\mathcal{N}(\mathbf{\Phi}(x_n)\mathbf{w},(q\mathbf{\eta}_n)^{-1}). \end{equation}

The pdf of the gamma distribution is $\text{Gamma}\left(\frac{\nu}{2},\frac{\nu}{2}-1\right)$ $\propto$ $\eta_n^{\frac{\nu}{2}-1}e^{-(\frac{\nu}{2}-1)\eta}$ and the pdf of Gaussian is $\mathcal{N}(\mathbf{\Phi}(x_n)\mathbf{w},(q\mathbf{\eta}_n)^{-1})$ $\propto$ $\eta_n^{\frac{1}{2}}e^{\frac{1}{2}((t_n-\mathbf{\Phi}(x_n)\mathbf{w})^Tq\eta_n(t_n-\mathbf{\Phi}(x_n)\mathbf{w}))}$. So, the posterior would be

\begin{equation} p(\eta_n|t_n,\nu)\text{ }\propto\text{ } \eta_n^{\frac{\nu}{2}-1}e^{-(\frac{\nu}{2}-1)\eta_n}\eta_n^{\frac{1}{2}}e^{\frac{1}{2}(s_nq\eta_ns_n)}=\eta_n^{\frac{\nu}{2}+\frac{1}{2}-1}e^{-(\frac{\nu}{2}-1)\eta_n+\frac{1}{2}(s_n^2q\eta_n)}=\eta_n^{\frac{\nu}{2}+\frac{1}{2}-1}e^{(\frac{\nu+qs_n^2}{2}-1)\eta_n}. \end{equation}

where $s_n=t_n-\mathbf{\Phi}(x_n)\mathbf{w}$. The last form is a Gamma$\left(\frac{\nu+1}{2},\frac{\nu+qs^2}{2}-1\right)$, so

\begin{equation} p(\eta_n|t_n,\nu)\text{ }\sim\text{ } \text{Gamma}(\frac{\nu+1}{2},\frac{\nu+qs^2}{2}-1). \end{equation}

$z_n|\textbf{w},\textbf{x}_n,t_n$ in the probit model:

$$p(z_n|\textbf{w},\textbf{x}_n,t_n)=\frac{p(t_n|\textbf{w},\textbf{x}_n,z_n)p(z_n|\textbf{w},\textbf{x}_n)}{\int p(t_n|\textbf{w},\textbf{x}_n,z_n)p(z_n|\textbf{w},\textbf{x}_n)dz_n}$$$$=\frac{[p(z_n>0|\textbf{w},\textbf{x}_n)^{t_n}(1-p(z_n>0|\textbf{w},\textbf{x}_n))^{1-t_n}]N(z_n|\Phi(\textbf{x}_n)^T\textbf{w},q)}{\int [p(z_n>0|\textbf{w},\textbf{x}_n)^{t_n}(1-p(z_n>0|\textbf{w},\textbf{x}_n))^{1-t_n}]N(z_n|\Phi(\textbf{x}_n)^T\textbf{w},q)dz_n}$$

$\textbf{z}_n|\textbf{x}_n,\mathbf{\mu}_{1:K},\textbf{Q}_{1:K}$ in the Gaussian mixture model:

$$p(\textbf{z}_n|\textbf{x}_n,\mathbf{\mu}_{1:K},\textbf{Q}_{1:K})=\frac{p(\textbf{x}_n|\textbf{z}_n,\mathbf{\mu}_{1:K},\textbf{Q}_{1:K})p(\textbf{z}_n)}{\sum_{j=1}^Kp(\textbf{x}_n|z_{nj}=1,\mathbf{\mu}_j,\textbf{Q}_j)p(z_{nj}=1)}$$$$=\frac{\Pi_{i=1}^K[\pi_ip(\textbf{x}_n|\textbf{z}_n,\mathbf{\mu}_i,\textbf{Q}_i^{-1})]^{z_{nk}}}{\sum_{j=1}^K\pi_jp(\textbf{x}_n|z_{nj}=1,\mathbf{\mu}_j,\textbf{Q}_j)}$$$$=\frac{\Pi_{i=1}^K[\pi_iN(\textbf{x}_n|\mathbf{\mu}_i,\textbf{Q}_i^{-1})]^{z_{nk}}}{\sum_{j=1}^K\pi_jN(\textbf{x}_n|\mathbf{\mu}_j,\textbf{Q}_j^{-1})}$$

$\textbf{z}_n|\textbf{x}_n,\textbf{W},\mathbf{\Sigma}$ in the factor model:

$$p(\textbf{z}_n|\textbf{x}_n,\textbf{W},\mathbf{\Sigma})=\frac{p(\textbf{x}_n|\textbf{z}_n,\textbf{W},\mathbf{\Sigma})p(\textbf{z}_n)}{p(\textbf{x}_n)}$$$$=\frac{N(\mathbf{x}_n|\mathbf{\mu}+\textbf{W}\textbf{z}_n,\mathbf{\Sigma})N(\textbf{0}_k,\textbf{I}_k)}{N(\mathbf{\mu},\textbf{W}\textbf{W}^T+\mathbf{\Sigma})}$$

which will be Gaussian distribution.

We define $Q(\mathbf{\theta},\mathbf{\theta}')=\mathtt{E}[\log(p(\mathbf{t},\mathbf{\eta}|\mathbf{\theta})|\mathbf{\theta}',\mathbf{x}]$. Applying conditional probabilities we get that

\begin{equation} Q(\mathbf{\theta},\mathbf{\theta}')=\mathtt{E}[\log(p(\mathbf{t}|\mathbf{\eta},\mathbf{\theta}))+\log p(\mathbf{\eta}|\mathbf{\theta})|\mathbf{\theta},\mathbf{x}]. \end{equation}

$p(\mathbf{\eta})$ only depends on $\nu$ and $\mathbf{\theta} '$, so it is constant over $\boldsymbol{\theta}$. Then, expanding the expectation we get

\begin{align} \mathtt{E}\log(p(\mathbf{t}|\boldsymbol{\eta},\mathbf{\theta})|\boldsymbol{\theta}',\mathbf{x}] & =\mathtt{E}\left[\frac{1}{2}\log q-\frac{q}{2}(\mathbf{t}-\mathbf{\Phi w})^T\text{diag}(\boldsymbol{\eta})(\mathbf{t}-\mathbf{\Phi w})\right]+C\\ & =\frac{N}{2}\log q-\frac{q}{2}(\mathbf{t}-\mathbf{\Phi w})^T\text{diag}(\mathtt{E}[\boldsymbol{\eta}|\mathbf{t},\mathbf{X},\boldsymbol{\theta}'](\mathbf{t}-\mathbf{\Phi w}))+C, \end{align}

where $p(\boldsymbol{\eta}|\mathbf{t},\nu)=\text{Gamma}(\frac{\nu+1}{2},\frac{\nu+q'(\mathbf{t}-\mathbf{\Phi w}')^2}{2}-1)$, so his expectation will be a vector with the elements (applying expectation of a Gamma distribution)

\begin{equation} \frac{\nu+1}{\nu+\frac{q'}{2}(t_n-\Phi(x_n)^T\mathbf{w}')^2-1} \end{equation}

mle_estimator_lm <- function (w, t_vector, phi) {
  if(is.vector(t_vector) == FALSE)
    t_vector <- as.vector(t_vector)
  if(is.matrix(phi) == FALSE)
    phi <- as.matrix(phi)
  M <- ncol(phi) 
  phi_w <- phi %*% w[1:M]
  Sig <- w[(M + 1):(M + 1)]
  sum(-(1 / 2) * log(2 * pi) - (1 / 2) * log(Sig ^ 2) - (1 / (2 * Sig ^ 2)) * (t_vector - phi_w) ^ 2)
}

mle_result <- function (mle_estimation_result, t_vector, phi) {
  phi        <- as.matrix(phi) 
  w          <- as.matrix(mle_estimation_result$par[-length(mle_estimation_result$par)])
  sigma      <- last(mle_estimation_result$par)
  variance   <- -solve(mle_estimation_result$hessian)
  w_se       <- sqrt(diag(variance))[1:length(w)]
  sigma_se   <- last(sqrt(diag(variance)))
  z_value    <- w/w_se
  p_value    <- 2 * (1 - pnorm(abs(z_value)))
  t_hat      <- phi %*% w 
  e          <- t_vector - t_hat
  e_st       <- e/sigma_se
  leverage   <- NULL 
  for(i in 1:nrow(phi)){
    leverage <- c(leverage, t(phi[i,]) %*% solve(t(phi) %*% phi) %*% phi[i,])
  }
  dev <- -2 * ((1 / 2) * log(2 * pi) - (1 / 2) * log(sigma ^ 2) - (1 / (2 * sigma ^ 2)) * (e ^ 2))
  # leverage is equivalent to diag(phi %*% solve(t(phi) %*% phi) %*% t(phi))
  # diagonal of the hat matrix
  return(list(w = w, sigma = sigma, variance = variance, w_se = w_se, sigma_se = sigma_se,
              z_value = z_value, p_value = p_value, t_hat = t_hat, e = e, e_st = e_st,
              leverage = leverage, dev = dev, n = nrow(phi)))
}

# EM algorithm functions
e_step <- function(t, X, q, w, nu) {
  eta <- (nu + 1) / (nu + q * (t - X %*% w) ^ 2 - 2)
  return(eta)
}

m_step <- function(t, X, q0, w0, eta) {
  w <- solve(t(X) %*% diag(as.vector(eta)) %*% X, t(X) %*% diag(as.vector(eta)) %*% t)
  q <- (N / ( t(t - X %*% w) %*% diag(as.vector(eta)) %*% (t - X %*% w)))
  return(list(w = as.vector(w), q = as.numeric(q)))
}

rob_reg <- function(t, X, q0, w0, nu, max_iter = 100) {
  q <- q0
  w <- w0
  q_conv      <- NULL
  e_conv      <- NULL
  loglik_conv <- NULL
  for(i in 1:max_iter) {
    eta    <- e_step(t, X, q, w, nu)
    wq     <- m_step(t, X, q, w, eta)
    q      <- wq$q
    w      <- wq$w
    loglik <- (length(t) / 2) * log(q) - q/2 * (t((t - X %*% w)) * diag(eta)) %*% (t - X %*% w)
    q_conv[i]      <- q
    e_conv[i]      <- sum((t - X %*% w) ^ 2)
    loglik_conv[i] <- loglik
  }
  e   <- as.vector(t - X %*% w)
  se  <- as.vector(sqrt(diag(solve(q * t(as.vector((nu + 1) * (nu - 2 - q * e ^ 2) / 
                                                   ((nu + q * e ^ 2 - 2) ^ 2)) * X) %*% X))))
  dev <- NULL 
  for(i in 1:length(t)){
    dev[i] <- -2 * ((1 / 2) * log(q) - q/2 * (t[i] - X[i,] %*% w) * diag(eta) * (t[i] - X[i,] %*% w))
  }
  return(list(w = w, q = q, eta = eta, e = e, se = se, e_conv = e_conv, q_conv = q_conv, 
              loglik_conv = loglik_conv, dev = dev))
}

rob_reg_plot <- function(rob_reg_results){
  ci_plot_data <- data.frame(coeff_number = seq(1:length(rob_reg_results$w))) %>% 
    bind_cols(data.frame(w = rob_reg_results$w)) %>% 
    mutate(lower_bound = w - 1.96 * rob_reg_results$se) %>% 
    mutate(upper_bound = w + 1.96 * rob_reg_results$se) %>% 
    mutate(color_ci = ifelse(upper_bound > 0 & lower_bound < 0, "red", "blue"))
  
  ci_plot <- ggplot(data = ci_plot_data, aes(x = as.factor(coeff_number), y = w)) + 
    geom_errorbar(aes(ymin = lower_bound, ymax = upper_bound), 
                  color = ci_plot_data$color_ci, width = 0.3) + 
    geom_point() + ggtitle("Robust Bayesian Estimator") +
    labs(x = "Coefficient", y = "w, w +/- 1.96 s.e.") + theme_excel()
  
  de_plot_data <- data.frame(observation_number = seq(1:length(rob_reg_results$dev))) %>% 
    bind_cols(data.frame(dev = rob_reg_results$dev)) %>% 
    mutate(color_de = ifelse(dev > as.numeric(quantile(rob_reg_results$dev, c(0.005, 0.995))[2]) | 
                             dev < as.numeric(quantile(rob_reg_results$dev, c(0.005, 0.995))[1]), "red","black"))
  
  de_plot <- ggplot(data = de_plot_data, aes(x = observation_number, y = dev)) + 
    geom_point(color = de_plot_data$color_de) +  
    geom_hline(yintercept =  quantile(rob_reg_results$dev, c(0.005, 0.995))[1]) +
    geom_hline(yintercept =  quantile(rob_reg_results$dev, c(0.005, 0.995))[2]) + 
    ggtitle("Robust Bayesian Estimator") +
    labs(x = "Observation", y = "Deviance Error") + theme_excel()
  
  tolerance <- 10 ^ -10
  conv_iter <- which((abs(diff(rob_reg_results$e_conv)) < tolerance & 
                      abs(diff(rob_reg_results$q_conv)) < tolerance &
                      abs(diff(rob_reg_results$loglik_conv)) < tolerance) == TRUE)[1]
  
  conv_plot_data <- data.frame(iter_number = seq(1:conv_iter), e_conv = rob_reg_results$e_conv[1:conv_iter], 
                               q_conv = rob_reg_results$q_conv[1:conv_iter],
                               loglik_conv = rob_reg_results$loglik_conv[1:conv_iter])
  
  q_conv_plot <- ggplot(data = conv_plot_data, aes(x = iter_number, y = q_conv)) +
    geom_point() + ggtitle("q convergence") +
    labs(x = "iteration", y = "q") + theme_excel()
  
  loglike_conv_plot <- ggplot(data = conv_plot_data, aes(x = iter_number, y = loglik_conv)) +
    geom_point() + ggtitle("loglik convergence") +
    labs(x = "iteration", y = "loglik") + theme_excel()
  
  e_conv_plot <- ggplot(data = conv_plot_data, aes(x = iter_number, y = e_conv)) +
    geom_point() + ggtitle("RSS convergence") +
    labs(x = expression(nu), y = "RSS") + theme_excel()
  
  return(list(ci_plot = ci_plot, de_plot = de_plot, q_conv_plot = q_conv_plot, 
              loglike_conv_plot = loglike_conv_plot, e_conv_plot = e_conv_plot))
}
